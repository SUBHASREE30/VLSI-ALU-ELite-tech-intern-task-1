module alu(
    input [3:0] A,
    input [3:0] B,
    input [2:0] opcode,
    output reg [3:0] out
);

  always @(*) begin
    case(opcode)
      3'b000: out = A + B;    // Addition
      3'b001: out = A - B;    // Subtraction
      3'b010: out = A & B;    // AND
      3'b011: out = A | B;    // OR
      3'b100: out = ~A;       // NOT A
      default: out = 4'b0000; // Default case
    endcase
  end
endmodule
